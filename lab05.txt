ZAD 2.

ZAD 3.

2. SELECT * FROM client WHERE TAX_IDENTIFIER IS NOT NULL ORDER BY SHORT_NAME ASC;

ZAD 4. 
1. SELECT DISTINCT FIRST_NAME FROM employee;
2. SELECT CONCAT(FIRST_NAME, ' - ', LAST_NAME) AS imie_nazwisko FROM employee WHERE LAST_NAME LIKE 'Ko%';
3. SELECT LAST_NAME, pensja FROM employee WHERE YEAR(DATE_OF_EMPLOYMENT) BETWEEN '2010' AND '2014' ;


ZAD 5.
